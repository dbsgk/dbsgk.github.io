## 9. Palindrome Number

// 내 꺼
class Solution {
    public boolean isPalindrome(int x) {
        if (x == 0) return true;
        String strX = "" + x;
        if (x < 0) return false;
        int a = x % 2; // 홀짝 구분
        int i = 0;
        int j = 0;
        
        if (a > 0) { //홀수
            j = (strX.length() -1) / 2;
            if ( x < 10) return true;
        } else {
            j = (strX.length()) / 2;
        }
        for (i = 0; i <= j; i++) {
            if (strX.charAt(i) != strX.charAt(strX.length() -1 -i)) { 
              return false;
            }
        } 
        return true;
        
    }
}

// leetcode 선생님꺼(arunkumar_hg)

class Solution {
    public boolean isPalindrome(int x) 
    {
        
        int reverse = 0;
        int temp = x;
        
        while(x!=0)
        {
            reverse = reverse*10+x%10;
            x = x/10;
        }
        return temp==reverse && temp>=0;      
        
    }
}
